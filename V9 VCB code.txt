        const plusInfosButtons = document.querySelectorAll('.plus-infos-btn');

        const inscrireButtons = document.querySelectorAll('.inscrire-btn');
        const inscrireSubpoleButtons = document.querySelectorAll('.inscrire-btn-subpole');

        plusInfosButtons.forEach(button => {
            button.addEventListener('click', () => {
                const pole = button.dataset.pole;
                const infosContainer = document.getElementById(`infos-${pole}`);
                if (infosContainer.style.display === 'none' || infosContainer.style.display === '') {
                    infosContainer.style.display = 'flex';
                } else {
                    infosContainer.style.display = 'none';
                }
            });
        });

        inscrireButtons.forEach(button => {
            button.addEventListener('click', () => {
                // Récupérer l'ID du conteneur parent
                const infosContainerId = button.closest('.infos-container').id;
                // Trouver l'élément d'inscription à l'intérieur de ce conteneur
                const inscriptionDiv = document.getElementById(`inscription-${infosContainerId.replace('infos-', '')}`);

                if (inscriptionDiv) {
                    if (inscriptionDiv.style.display === 'none' || inscriptionDiv.style.display === '') {
                        inscriptionDiv.style.display = 'block';
                    } else {
                        inscriptionDiv.style.display = 'none';
                    }
                }
            });
        });

        inscrireSubpoleButtons.forEach(button => {
            button.addEventListener('click', () => {
                // Récupérer l'ID du conteneur parent
                const infosContainerId = button.closest('.infos-container').id;
                // Trouver l'élément d'inscription à l'intérieur de ce conteneur
                const inscriptionDiv = document.getElementById(`inscription-${infosContainerId.replace('infos-', '')}`);

                if (inscriptionDiv) {
                    if (inscriptionDiv.style.display === 'none' || inscriptionDiv.style.display === '') {
                        inscriptionDiv.style.display = 'block';
                    } else {
                        inscriptionDiv.style.display = 'none';
                    }
                }
            });
        });

        function ajouterInscrit(pole) {
            const prenomInput = document.getElementById(`prenom-${pole}`);
            const listeInscrits = document.getElementById(`liste-inscrits-${pole}`);

            if (prenomInput.value.trim() !== "") {
                const listItem = document.createElement("li");
                listItem.textContent = prenomInput.value;
                listeInscrits.appendChild(listItem);
                prenomInput.value = ""; // Clear the input
            } else {
                alert("Veuillez entrer votre prénom avant de valider.");
            }
        }
    </script>
